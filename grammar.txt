Program -> Statement Program'
Program' -> Program
        | epsilon

Statement -> Let
	         | Instr

Instr -> Expr ;

Let -> let Identifier Params = Expr end

Params -> Identifier Params
       | epsilon

Expr -> Expr Argument
      | Literal
      | List //not implemented yet
      | Identifier
      | let Identifier = Expr in Expr //not implemented yet
      | ( Expr )
      | begin Expr end

Literal -> Number //not Implemented yet
        | Text
        | String //not implemented yet

Argument -> Expr Argument
         | epsilon

List -> [ Expr RestList ]
        | [ ]

RestList -> ; Expr RestList'
RestList' -> RestList
          | epsilon

Text -> text

Number -> [0-9] Number
        | epsilon

Identifier -> [_A-Za-z] Identifier
           | [_A-Za-z] [_A-Za-z0-9] Identifier
           | epsilon
